Rails Test Directory Structure
ROR 4.2
#
test             - tests, fixtures, other test apparatus
│                  run by: rake
│                  generated by $rails new ap
│
├─controllers    - functional tests of controllers;
│ │                run by: rake, rake test:functionals, rake test:controllers
│ │                generated by $rails new ap
│ │
│ └─controllername_controller_test.rb
│                  inherits from ActionController::TestCase
│                  generated by ap$rails generate controller controllername
│
├─fixtures       - fixtures
│ │                generated by $rails new ap
│ │
│ └─modelnames.yml
│                  generated by ap$rails generate model Modelname
│
├─functional     - miscellaneous functional tests
│                  run by: rake, rake test:functionals
│
├─helpers        - unit tests of helpers
│ │                run by: rake, rake test:units, rake test:helpers
│ │                generated by $rails new ap
│ │
│ └─controllername_helper_test.rb
│                  inherits from ActionView::TestCase
│                  To test the output of the helper's methods,
│                  you must mixin the helper: include ControllerNameHelper
│                  generated by ap$rails generate controller controllername
│
├─integration    - integration tests
│ │                run by: rake, rake test:integration
│ │                generated by $rails new ap
│ │
│ └─integration_test_name_test.rb
│                  inherits from ActionDispatch::IntegrationTest
│                  generated by:
│                    ap$rails generate integration_test integration_test_name
│
├─jobs           - job tests
│ │                run by: rake, rake test:jobs
│ │                generated by ap$rails generate job jobname
│ │
│ └─jobname_test.rb
│                  inherits from ActiveJob::TestCase
│                  generated by ap$rails generate job jobname
│
├─mailers        - functional tests of mailers
│ │                run by: rake, rake test:functionals, rake test:mailers
│ │                generated by ap$rails generate mailer MailerName
│ │
│ ├-mailer_name_test.rb
│ │                inherits from ActionMailer::TestCase
│ │                generated by ap$rails generate mailer MailerName
│ │
│ └─preview      - ?
│                  inherits from ActionMailer::Preview
│                  not run by any rake task
│                  generated by ap$rails generate mailer MailerName
│
├─models         - unit tests of models
│ │                run by: rake, rake test:units, rake test:models
│ │                generated by $rails new ap
│ │
│ └─modelname_test.rb
│                  inherits from ActiveSupport::TestCase
│                  generated by ap$rails generate model Modelname
│
├─unit           - miscellaneous unit tests
│                  run by rake, rake test:units
│
└─test_helper.rb - helper for all tests
                   include in all test files
                   generated by $rails new ap
